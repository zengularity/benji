BenjiFromContext 

abstract class BenjiFromContext extends api.BuiltInComponentsFromContext with BenjiComponentsWithInjector

Can be used for a custom application loader.

import play.api.ApplicationLoader

class MyApplicationLoader extends ApplicationLoader {
  def load(context: ApplicationLoader.Context) =
    new MyComponents(context).application
}

class MyComponents(context: ApplicationLoader.Context)
    extends BenjiFromContext(context) {
  lazy val router = play.api.routing.Router.empty
}
